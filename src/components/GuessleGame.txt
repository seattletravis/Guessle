import { useEffect, useRef } from 'react';
import * as THREE from 'three';
import * as CANNON from 'cannon-es';

export default function GuessleGame() {
	const canvasRef = useRef<HTMLCanvasElement | null>(null);

	useEffect(() => {
		const scene = new THREE.Scene();

		const camera = new THREE.PerspectiveCamera(
			45,
			window.innerWidth / window.innerHeight,
			0.1,
			1000,
		);
		camera.position.set(5, 2, 20);

		const renderer = new THREE.WebGLRenderer({
			canvas: canvasRef.current!,
			antialias: false,
			stencil: false,
			depth: true,
			powerPreference: 'high-performance',
		});

		renderer.setSize(window.innerWidth, window.innerHeight);
		renderer.setPixelRatio(window.devicePixelRatio);

		renderer.shadowMap.enabled = false;
		renderer.toneMapping = THREE.NoToneMapping;
		renderer.outputColorSpace = THREE.LinearSRGBColorSpace;

		const world = new CANNON.World({
			gravity: new CANNON.Vec3(0, -9.82, 0),
		});
		// --- JAR PHYSICS ---
		const jarBody = new CANNON.Body({
			type: CANNON.Body.STATIC,
		});

		// Cylinder walls
		const wallShape = new CANNON.Cylinder(5, 5, 10, 32);
		const wallQuat = new CANNON.Quaternion();
		wallQuat.setFromEuler(Math.PI / 2, 0, 0);
		jarBody.addShape(wallShape, new CANNON.Vec3(0, 0, 0), wallQuat);

		// Bottom (thin cylinder)
		const bottomShape = new CANNON.Cylinder(5, 5, 0.2, 32);
		const bottomQuat = new CANNON.Quaternion();
		bottomQuat.setFromEuler(Math.PI / 2, 0, 0);
		jarBody.addShape(bottomShape, new CANNON.Vec3(0, -5, 0), bottomQuat);

		world.addBody(jarBody);
		// --TABLE-- Physics
		const tableBody = new CANNON.Body({
			type: CANNON.Body.STATIC,
			shape: new CANNON.Box(new CANNON.Vec3(15, 0.5, 15)),
			position: new CANNON.Vec3(0, -5.5, 0),
		});
		world.addBody(tableBody);

		// --LIGHTS -- Add a basic light so we can see geometry
		const ambient = new THREE.AmbientLight(0xffffff, 0.6);
		scene.add(ambient);

		const directional = new THREE.DirectionalLight(0xffffff, 0.8);
		directional.position.set(10, 20, 10);
		scene.add(directional);

		// --- JAR GROUP ---
		const jarGroup = new THREE.Group();
		scene.add(jarGroup);

		// Glass material
		const glassMat = new THREE.MeshStandardMaterial({
			color: 0xffffff,
			transparent: true,
			opacity: 0.25,
		});

		// Jar walls
		const wallGeo = new THREE.CylinderGeometry(5, 5, 10, 32, 1, true);
		const wallMesh = new THREE.Mesh(wallGeo, glassMat);
		wallMesh.position.set(0, 0, 0);
		jarGroup.add(wallMesh);

		// Jar bottom
		const bottomGeo = new THREE.CircleGeometry(5, 32);
		const bottomMat = new THREE.MeshStandardMaterial({ color: 0xdddddd });
		const bottomMesh = new THREE.Mesh(bottomGeo, bottomMat);
		bottomMesh.rotation.x = -Math.PI / 2;
		bottomMesh.position.y = -5;
		jarGroup.add(bottomMesh);

		// Jar rim
		const rimGeo = new THREE.TorusGeometry(5, 0.3, 16, 100);
		const rimMat = new THREE.MeshStandardMaterial({ color: 0xffffff });
		const rimMesh = new THREE.Mesh(rimGeo, rimMat);
		rimMesh.position.y = 5;
		rimMesh.rotation.x = Math.PI / 2;
		jarGroup.add(rimMesh);

		//  --OBJECTS-- Add temp test objects for light confirmation
		const testGeo = new THREE.BoxGeometry(2, 2, 2);
		const testMat = new THREE.MeshStandardMaterial({ color: 'hotpink' });
		const testMesh = new THREE.Mesh(testGeo, testMat);
		scene.add(testMesh);
		// Add background
		scene.background = new THREE.Color(0x222222);
		// Add Table
		const tableGeo = new THREE.BoxGeometry(30, 1, 30);
		const tableMat = new THREE.MeshStandardMaterial({ color: 0x8b4513 });
		const table = new THREE.Mesh(tableGeo, tableMat);
		table.position.set(0, -5, 0);
		scene.add(table);

		function animate() {
			requestAnimationFrame(animate);
			world.step(1 / 60);
			renderer.render(scene, camera);
			jarGroup.position.copy(jarBody.position as any);
			jarGroup.quaternion.copy(jarBody.quaternion as any);
		}
		animate();

		return () => {
			renderer.dispose();
		};
	}, []);

	return (
		<div style={{ width: '100vw', height: '100vh' }}>
			<canvas ref={canvasRef} />
		</div>
	);
}
